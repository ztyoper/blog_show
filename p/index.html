	public static void quick_sort(int[] q, int l, int r) {
		if (l >= r)

			do j--; while(q[j] > x);
			if(i<j) swap(q, i, j);
		}
		quick_sort(q, l, j);
		quick_sort(q, j+1, r);
	}
	private static void swap(int[] q, int i , int j) {
		int t = q[i];
		q[i] = q[j];
		q[j] = t;
	}

}
