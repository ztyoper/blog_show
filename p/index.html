	public static void quick_sort(int[] q, int l, int r) {
		if (l >= r)
			return;
        // ÂèØ‰ª•ÁêÜËß£‰∏∫Èöè‰æøÊâæ‰∏Ä‰∏™Âü∫ÂáÜÁÇπÔºåÁÑ∂ÂêéÂ∞ÜÂå∫Èó¥Âèò‰∏∫Â∑¶ËæπÊòØÂÖ®ÈÉ®Â∞è‰∫éÁ≠â‰∫éÂü∫ÂáÜÁÇπÔºåÂè≥ËæπÊòØÂÖ®ÈÉ®Â§ß‰∫éÁ≠â‰∫éÂü∫ÂáÜÁÇπ

		while(i<j) {
			do i++; while(q[i] < x);
			do j--; while(q[j] > x);
			if(i<j) swap(q, i, j);
		}
		quick_sort(q, l, j);
		quick_sort(q, j+1, r);
	}
	private static void swap(int[] q, int i , int j) {
		int t = q[i];
		q[i] = q[j];
		q[j] = t;
	}
